```cpp
#define MAXN   (11)
#define MAXO   (150'001)
#define SIZEH  (1 << 16)

struct Kitchen {
    int t;			        // next time of food complete
    int cooktime;	// cooking time
    int qs;			// que index for start 
    int qe;			// que index for end 
    int queue[MAXO];
};
int N;
Kitchen kits[MAXN];

struct Order {
    int inext;		// hash
    int mid;		// mID
    int t;			// time
    int ct;		// canceled time
    int m;		// # of food remained
    int lcnt;		// # of food done	
    int list[MAXN - 1];
};

int ocnt;
Order orders[MAXO];
int hash[SIZEH];
int numorder;

Order& getOrder(int mID) {
	int hidx = mID & (SIZEH - 1);
	int oidx = hash[hidx];

//for ( ; oidx && mID != orders[oidx].mid; oidx = orders[oidx].inext);
	while (oidx && mID != orders[oidx].mid)
		oidx = orders[oidx].inext;
	return orders[oidx];
}

void run(int mTime) {
	for (int k = 1; k <= N; ++k) {
		Kitchen& kit = kits[k];
		int& t = kit.t;
		int cooktime = kit.cooktime;
		int& qs = kit.qs;
		int& qe = kit.qe;
		int* que = kit.queue;

		while (qs != qe) {
			Order& order = orders[que[qs]];
			if (order.ct) {
				if (t < order.ct) {
					order.list[order.lcnt++] = k;
					t += cooktime;
				}
			}
			else {
				if (mTime < t) break;
				order.list[order.lcnt++] = k;

				if (--order.m == 0) --numorder;
				t += cooktime;
			}
			if (++qs == qe) t = 0;
		}
	}
}

void init(int N, int mCookingTimeList[])
{
	::N = N;
	ocnt = 1;
	numorder = 0;

	// set kitchen
	for (int k = 1; k <= N; ++k) {
		Kitchen& kit = kits[k];
		kit.t = 0;
		kit.cooktime = mCookingTimeList[k - 1];
		kit.qs = kit.qe = 0;
	}

	// init hash
	for (int h = 0; h < SIZEH; ++h)
		hash[h] = 0;
}

int order(int mTime, int mID, int M, int mDishes[])
{
	// catch up
	run(mTime);

	// add order
	Order& order = orders[ocnt];
	order.mid = mID;
	order.t = mTime;
	order.ct = 0;
	order.m = M;
	order.lcnt = 0;

	// hash table
	int hidx = mID & (SIZEH - 1);
	order.inext = hash[hidx];
	hash[hidx] = ocnt;

	// add order to kitchen queue
	for (int i = 0; i < M; ++i) {
		Kitchen& kit = kits[mDishes[i]];
		kit.queue[kit.qe++] = ocnt;
		if (kit.t == 0)
			kit.t = mTime + kit.cooktime;
	}
	++ocnt;
	return ++numorder;
}

int cancel(int mTime, int mID)
{
	// catch up
	Order& order = getOrder(mID);
	order.ct = mTime;
	run(mTime);

	// rearrange next que
	for (int i = 0; i < order.lcnt; ++i) {
		int k = order.list[i];
		Kitchen& kit = kits[k];
		int& qs = kit.qs;
		int& qe = kit.qe;
		int* que = kit.queue;
		bool isalloc = false;

		while (qs != qe && !isalloc) {
			Order& norder = orders[que[qs]];
			if (!norder.ct) {
				norder.list[norder.lcnt++] = k;
				if (--norder.m == 0) --numorder;
				isalloc = true;
			}
			if (++qs == qe) kit.t = 0;
		}
	}
	return --numorder;
}

int getStatus(int mTime, int mID)
{
	// catch up
	run(mTime);
	Order& order = getOrder(mID);
	return order.ct? -1: order.m;
}
```
